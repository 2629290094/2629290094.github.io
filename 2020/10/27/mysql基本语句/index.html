单表子句-from

select 列1,列2 from 表
select * from 表(不要对大表操作)

单表子句-where

–where配合等值查询

select 列1,列2 from 表 where 列n=条件;

–where配合比较操作符

select 列1,列2 from 表 where 列n <(>,!=,>=,<=) 条件

–where配合逻辑运算符

select 列1,列2 from 表 where 列n 条件 and(or) 列m 条件

–where配合模糊查询

select * from 表 where 列n like 'qwq%';(百分号不能放前面)

–where配合in语句

select * from 表 where 列n in(条件1,条件2);

–where配合between and

select * from 表 where 列n between num1 and num2;

单表子句-group by

例子1.统计city表每个国家的总人口数

select countrycode,sum(population) 
from city
group by countrycode

例子2.统计中国,每个省的总人口数

select district,sum(population)
from city 
where countrycode = "CHN"
group by district;

例子3.统计世界上,每个国家的城市数量

select countrycode count(id)
from city
group by countrycode

例子4.统计中国,每个省的城市列表

select district,group_concat(NAME)
from city
where countrycode = 'CHN'
group by district;

单表子句-having

例子1.统计中国,每个省的总人口数,只打印总人口数小于100W的信息

select district,sum(population)
from city 
where countrycode = "CHN"
group by district;
having sum(population)<10000000

注意:having 条件是不走索引的.

单表子句-order by

例子1.查询中国的城市信息,并按人口数量从大到小进行排序

select * from city
where countrycode = 'CHN'
order by population desc;

例子2.统计中国每个省的总人口,找出大于500w的,并按总人口从大到小排序

select district sum(population)
from city
where countrycode = 'CHN'
group by district
having population > 5000000
order by desc;

单表语句-limit

例子1.统计中国每个省的总人口,找出大于500w的,并按总人口从大到小排序,只显示前三名

select district sum(population)
from city
where countrycode = 'CHN'
group by district
having population > 5000000
order by desc;
limit 3;

limit n,m
跳过前n名,再显示前m名

多表链接查询

内链接（两表某字段的交集）

select s.Sno,s.cno,s.score,c.cname 
from sc as s inner join course as c on s.cno=c.cno　order by c.cno,s.Sno;

左链接（以左边为基准匹配，未找到为ＮＵＬＬ）

select s.Sno,s.cno,s.score,c.cname 
from sc as s left join course as c on s.cno=c.cno　order by c.cno,s.Sno
having s.cno is NULL;

自关联

自己关联自己的表（省市县，上下级）

select c.name,i.name from city as c inner join city as i 
on c.id = i.pid order by c.id;

子查询

select里嵌套select
select * from city where pid = (select id from city where name='北京');

外键

alter table goods add foreign key(brand_id) references goods_brands(id);
